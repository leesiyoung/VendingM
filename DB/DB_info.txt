product(상품 정보)
	name(품명) varchar(20)
	code(코드) int unsigned primary
	price(가격) int unsigned




stock_001(001자판기 재고)
	p_code(코드) int unsigned foreign key(product.code)
	stock(재고량) int unsigned
	mf_date(제조일자) date
	ep_date(유통기한) date




stock_002(002자판기 재고)
	p_code(코드) int unsigned foreign key(product.code)
	stock(재고량) int unsigned
	mf_date(제조일자) date
	ep_date(유통기한) date




sales(판매량)
	p_code(코드) int unsigned foreign key(product.code)
	sold(판매일) date




bill(권액별 재고)
	currency(권액 종류) int unsigned
	stock(매수) int unsigned
